# positive-num
python program to print all positive numbers in range
# list of numbers 

list1 = [12,-7,5,64,-14] 
# iterating each number in list 

for num in list1: 
    # checking condition 

    if num >= 0: 

       print(num, end = " ")
       
 # output : 12,5,64
